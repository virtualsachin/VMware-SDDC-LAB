Examples
--------

.. code-block:: yaml

    - hosts: localhost
      connection: local
      collections:
        - vmware.alb
      vars:
        avi_credentials:
          username: "avi_user"
          password: "avi_password"
          controller: "192.168.138.18"
          api_version: "21.1.1"
      tasks:
        - name: Example to create ApiFileservice object
          avi_api_fileservice:
            avi_credentials: "{{ avi_credentials }}"
            state: present
            name: sample_api_fileservice
